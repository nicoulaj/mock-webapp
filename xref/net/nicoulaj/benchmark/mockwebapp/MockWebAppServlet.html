32  /**
33   * {@link HttpServlet} that defers requests handling to a {@link MockWebAppConfig}.
34   *
35   * @author Julien Nicoulaud <julien.nicoulaud@gmail.com>
36   * @since 1.0.0
37   */
38  public class MockWebAppServlet extends HttpServlet {
39  
40      /**
41       * The name of property used to define the path to the configuration file to use.
42       */
43      public static final String MOCK_WEB_APP_CONF_PROPERTY = "mock-web-app-conf";
44  
45      /**
46       * The {@link MockWebAppConfig} currently in use.
47       */
48      protected MockWebAppConfig mockWebAppConfig;
49  
50      /**
51       * The file monitor used to detect changes in the configuration file.
52       */
53      protected DefaultFileMonitor fileMonitor;
54  
55      /**
56       * Initialize the servlet.
57       * <p/>
58       * Loads the {@link #mockWebAppConfig} by looking up the {@link #MOCK_WEB_APP_CONF_PROPERTY} as a system property or init parameter,
59       * and sets up a listener on the file changes.
60       *
61       * @param config the {@link ServletConfig}, optionally with a {@link #MOCK_WEB_APP_CONF_PROPERTY} parameter.
62       * @throws ServletException if the {@link #MOCK_WEB_APP_CONF_PROPERTY} was neither defined as a system property nor an init parameter.
63       */
64      @Override
65      public void init(ServletConfig config) throws ServletException {
66          super.init(config);
67  
68          // Locate the config file parameter.
69          String configFilePath = System.getProperty(MOCK_WEB_APP_CONF_PROPERTY);
70          if (configFilePath == null) configFilePath = config.getInitParameter(MOCK_WEB_APP_CONF_PROPERTY);
71          if (configFilePath == null) throw new ServletException("No mock web app config file defined. Please define one using the '" + MOCK_WEB_APP_CONF_PROPERTY + "' system property or servlet init parameter.");
72  
73          // Setup the config file changes listener.
74          try {
75              fileMonitor = new DefaultFileMonitor(new ConfigFileListener());
76              fileMonitor.addFile(VFS.getManager().resolveFile(new File("."), configFilePath));
77              fileMonitor.start();
78          } catch (Exception e) {
79              getServletContext().log("Failed setting up config file changes listener, config file changes will not be taken into account", e);
80          }
81  
82          // Load the config from the file.
83          final File configFile = new File(configFilePath);
84          try {
85              mockWebAppConfig = MockWebAppConfig.Parser.parseConfig(configFile);
86              getServletContext().log("Loaded config file " + configFilePath);
87          } catch (Exception e) {
88              getServletContext().log("Failed loading config, please replace it with a valid one", e);
89          }
90      }
91  
92      /**
93       * Prepare the servlet for stopping service.
94       * <p/>
95       * Disposes the config file changes monitor.
96       */
97      @Override
98      public void destroy() {
99          if (fileMonitor != null) fileMonitor.stop();
100         super.destroy();
101     }

215     /**
216      * {@link FileListener} handling the mock webapp config file changes.
217      *
218      * @author Julien Nicoulaud <julien.nicoulaud@gmail.com>
219      * @see MockWebAppServlet#init(javax.servlet.ServletConfig)
220      * @see MockWebAppServlet#fileMonitor
221      * @since 1.0.0
222      */
223     protected class ConfigFileListener implements FileListener {
224 
225         /**
226          * Handle a file creation event.
227          * <p/>
228          * Defers the event to {@link #fileChanged(org.apache.commons.vfs.FileChangeEvent)}.
229          *
230          * @param fileChangeEvent the event to handle.
231          */
232         public void fileCreated(FileChangeEvent fileChangeEvent) {
233             fileChanged(fileChangeEvent);
234         }
235 
236         /**
237          * Handle a file deletion event.
238          * <p/>
239          * Unsets {@link MockWebAppServlet#mockWebAppConfig}.
240          *
241          * @param fileChangeEvent the event to handle.
242          */
243         public void fileDeleted(FileChangeEvent fileChangeEvent) {
244             getServletContext().log("Config file deleted");
245             mockWebAppConfig = null;
246         }
247 
248         /**
249          * Handle a file change event.
250          * <p/>
251          * Updates {@link MockWebAppServlet#mockWebAppConfig}.
252          *
253          * @param fileChangeEvent the event to handle.
254          */
255         public void fileChanged(FileChangeEvent fileChangeEvent) {
256             try {
257                 final File configFile = new File(fileChangeEvent.getFile().getURL().getFile());
258                 mockWebAppConfig = MockWebAppConfig.Parser.parseConfig(configFile);
259                 getServletContext().log("Updated config from " + configFile.getPath());
260             } catch (Exception e) {
261                 getServletContext().log("Failed updating config", e);
262             }
263         }
264     }
