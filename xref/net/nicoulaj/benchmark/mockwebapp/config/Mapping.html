16  package net.nicoulaj.benchmark.mockwebapp.config;
17  
18  import net.nicoulaj.benchmark.mockwebapp.config.then.DelayAction;
19  import net.nicoulaj.benchmark.mockwebapp.config.then.RandomDelayAction;
20  import net.nicoulaj.benchmark.mockwebapp.config.then.StatusAction;
21  import net.nicoulaj.benchmark.mockwebapp.config.then.ThenStatement;
22  import net.nicoulaj.benchmark.mockwebapp.config.when.*;
23  
24  import javax.servlet.http.HttpServletRequest;
25  import javax.servlet.http.HttpServletResponse;
26  import javax.xml.bind.annotation.*;
27  import java.util.List;
28  
29  /**
30   * A {@link Mapping} is:
31   * <ul>
32   * <li>a set of {@link WhenStatement}s defining what requests should be matched</li>
33   * <li>a set of {@link ThenStatement}s defining how matching requests should be processed</li>
34   * </ul>
35   *
36   * @author Julien Nicoulaud <julien.nicoulaud@gmail.com>
37   * @since 1.0.0
38   */
39  @XmlAccessorType(XmlAccessType.FIELD)
40  public class Mapping implements ConfigFragment {
41  
42      /**
43       * The list of {@link WhenStatement}s defining what requests should be matched.
44       *
45       * @see net.nicoulaj.benchmark.mockwebapp.config.when
46       */
47      @XmlElementWrapper(name = "when", required = true, nillable = false)
48      @XmlElements({
49                           @XmlElement(name = "url", type = URLCondition.class),
50                           @XmlElement(name = "header", type = HeaderCondition.class),
51                           @XmlElement(name = "method", type = MethodCondition.class),
52                           @XmlElement(name = "remote-address", type = RemoteAddressCondition.class)
53                   })
54      public List<WhenStatement> whenStatements;
55  
56      /**
57       * The list of {@link ThenStatement}s defining how matching requests should be processed.
58       *
59       * @see net.nicoulaj.benchmark.mockwebapp.config.then
60       */
61      @XmlElementWrapper(name = "then", required = true, nillable = false)
62      @XmlElements({
63                           @XmlElement(name = "status", type = StatusAction.class),
64                           @XmlElement(name = "delay", type = DelayAction.class),
65                           @XmlElement(name = "random-delay", type = RandomDelayAction.class)
66                   })
67      public List<ThenStatement> thenStatements;
68  
69      /**
70       * Assert this {@link Mapping} is valid.
71       * <p/>
72       * Checks everything that cannot be enforced through the XML schema.
73       *
74       * @throws Throwable if an element of the {@link Mapping} is invalid.
75       */
76      public void validate() throws Throwable {
77          assert whenStatements != null : "\"when\" statements list is missing";
78          assert thenStatements != null : "\"then\" statements list is missing";
79          assert !whenStatements.isEmpty() : "\"when\" statements list is empty";
80          assert !thenStatements.isEmpty() : "\"then\" statements list is empty";
81          for (WhenStatement whenStatement : whenStatements) whenStatement.validate();
82          for (ThenStatement thenStatement : thenStatements) thenStatement.validate();
83      }
84  
85      /**
86       * Test whether the given {@link HttpServletRequest} matches this mapping.
87       *
88       * @param req the {@link HttpServletRequest} to test.
89       * @return true if if the request matches all {@link WhenStatement}s in {@link #whenStatements}.
90       */
91      public boolean matches(HttpServletRequest req) {
92          for (WhenStatement matcher : whenStatements) if (!matcher.matches(req)) return false;
93          return true;
94      }
95  
96      /**
97       * Process the given request.
98       * <p/>
99       * Calls {@link ThenStatement#process(javax.servlet.http.HttpServletRequest, javax.servlet.http.HttpServletResponse)} for
100      * each {@link ThenStatement} in {@link #thenStatements}.
101      *
102      * @param req  the {@link HttpServletRequest} to process.
103      * @param resp the {@link HttpServletResponse} to use.
104      */
105     public void process(HttpServletRequest req, HttpServletResponse resp) {
106         for (ThenStatement processor : thenStatements) processor.process(req, resp);
107     }
108 }
