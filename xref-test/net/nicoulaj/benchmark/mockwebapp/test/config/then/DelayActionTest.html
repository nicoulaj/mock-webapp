30  import static org.testng.Assert.assertTrue;

32  /**
33   * Tests for {@link DelayAction}.
34   *
35   * @author Julien Nicoulaud <julien.nicoulaud@gmail.com>
36   * @since 1.0.0
37   */
38  public class DelayActionTest extends AbstractMockWebAppTest {
39  
40      /**
41       * The safety margin to use when calculating the time elapsed.
42       */
43      protected static final int TOLERANCE = 50;
44  
45      /**
46       * Provide a list of coherent delays.
47       * <p/>
48       * Can be used by tests method with a (int delay) signature.
49       *
50       * @return an {@link Iterator} of delays as single argument.
51       */
52      @DataProvider
53      public Iterator<Object[]> delayRangesDataProvider() {
54          final List<Object[]> params = new ArrayList<Object[]>();
55          for (int i : new int[]{5, 50, 100, 300, 1000}) params.add(new Object[]{i});
56          return params.iterator();
57      }
58  
59      /**
60       * Assert the time elapsed for processing the request is the right one.
61       *
62       * @param delay the delay to use for the test.
63       * @throws IOException  should never happen.
64       * @throws SAXException should never happen.
65       */
66      @Test(dataProvider = "delayRangesDataProvider")
67      public void elapsedTimeShouldEqualDelay(int delay) throws IOException, SAXException {
68  
69          final DelayAction stmt = new DelayAction();
70          stmt.time = delay;
71  
72          final GetMethodWebRequest webRequest = new GetMethodWebRequest("http://localhost/test");
73          final long startTime = System.currentTimeMillis();
74          stmt.process(getRequest(webRequest), getResponse(webRequest));
75          final long elapsedTime = System.currentTimeMillis() - startTime;
76  
77          assertTrue(elapsedTime >= stmt.time);
78          assertTrue(elapsedTime <= stmt.time + TOLERANCE);
79      }
80  
81      /**
82       * Assert validating a {@link DelayAction} with a negative delay setting throws an error.
83       *
84       * @throws Throwable should always happen.
85       */
86      @Test(expectedExceptions = Throwable.class,
87            expectedExceptionsMessageRegExp = "The delay cannot be negative")
88      public void delayShouldBePositive() throws Throwable {
89          final DelayAction stmt = new DelayAction();
90          stmt.time = -1;
91          stmt.validate();
92      }
93  }
